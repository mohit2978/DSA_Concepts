```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean isSameTree(TreeNode p, TreeNode q) {
        if(p==null&&q==null) return true;
        if(p!=null&&q==null) return false;
        if(p==null&&q!=null) return false;
        
        boolean lres=isSameTree(p.left,q.left);
        boolean rres=isSameTree(p.right,q.right);
        
        boolean myres=(p.val==q.val);
        return lres&&rres&&myres;
        
    }
}
```

```java
/**

 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public List<List<Integer>> levelOrderBottom(TreeNode root) {
        List<List<Integer>> res = new ArrayList<List<Integer>>();
        if (root == null) return res;
        LinkedList <TreeNode>q=new LinkedList<>();
        q.addLast(root);
        while(q.size()>0){
            List<Integer>temp=new ArrayList<>();
            int sz=q.size();
            while(sz-->0){
                TreeNode fr=q.removeFirst();
                temp.add(fr.val);
                if(fr.left!=null)
                    q.addLast(fr.left);
                if(fr.right!=null)
                    q.addLast(fr.right);
            }
        res.add(temp);    
        }
        Collections.reverse(res);
        return res;
        
    }
}
```

 ```cpp
        auto r=check(node->right);
        pair<bool,int> res;
        res.first=l.first & r.first & abs(l.second-r.second)<=1;
        res.second=max(l.second,r.second)+1;
        return res;

    }

        if(root==nullptr) return true;
        return check(root).first;
```

 ```cpp
pair<bool, int> check(TreeNode* node) {
    if (node == nullptr) return {true, 0};

    auto l = check(node->left);
    if (!l.first) return {false, 0}; // Early exit: Left is unbalanced

    auto r = check(node->right);
    if (!r.first) return {false, 0}; // Early exit: Right is unbalanced

    bool balanced = abs(l.second - r.second) <= 1;
    int height = max(l.second, r.second) + 1;

    return {balanced, height};
}
 ```

 ```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public class nodepair{
        TreeNode child;
        int par;
        nodepair(){
            par=-1;
            }
        nodepair(TreeNode child,int par){
            this.child=child;
            this.par=par;
        }
    }
    public boolean isCousins(TreeNode root, int x, int y) {
         LinkedList <nodepair>q=new LinkedList<>();
        nodepair rt=new nodepair(root,-1);
        q.addLast(rt);
        while(q.size()>0){
            int sz=q.size();
            nodepair xchck=new nodepair();
            nodepair ychck=new nodepair();
            while(sz-->0)
            {
                nodepair fr=q.removeFirst();
                if(fr.child!=null&&fr.child.val==x){
                    xchck.child=fr.child;
                    xchck.par=fr.par;
                    if(ychck.par!=-1){
                       return ychck.par!=xchck.par;
                    }
                }
                if(fr.child!=null&&fr.child.val==y){
                    ychck.child=fr.child;
                    ychck.par=fr.par;
                    if(xchck.par!=-1){
                       return ychck.par!=xchck.par;
                    }
                }
                if(fr.child!=null&&fr.child.left!=null){
                    nodepair lc=new nodepair(fr.child.left,fr.child.val);
                    q.addLast(lc);
                }
                    
                if(fr.child!=null&&fr.child.right!=null)
                { nodepair rc=new nodepair(fr.child.right,fr.child.val);
                    q.addLast(rc);
                    
                }
                
            }

        }
        return false;
    }
}
```

  ```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }

 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int res=(int)-1e6;
    private void helper(TreeNode root,int mn,int mx){
        if(root==null) return;
        int t1=Math.abs(mn-root.val);
        int t2=Math.abs(mx-root.val);
        res=Math.max(t1,Math.max(t2,res));
        mn=Math.min(mn,root.val);
        mx=Math.max(mx,root.val);
        helper(root.left,mn,mx);
        helper(root.right,mn,mx);
        
    }
    public int maxAncestorDiff(TreeNode root) {
        int mn=root.val;
        int mx=root.val;
        helper(root,mn,mx);
        return res;
    }
}
  ```

## LC 2196. Create Binary Tree From Descriptions


### **Problem Statement**

You are given a 2D integer array `descriptions` where `descriptions[i] = [parent_i, child_i, isLeft_i]` indicates that `parent_i` is the parent of `child_i` in a binary tree of unique values. Furthermore:

* If `isLeft_i == 1`, then `child_i` is the left child of `parent_i`.
* If `isLeft_i == 0`, then `child_i` is the right child of `parent_i`.

Construct the binary tree described by `descriptions` and return its root.

*The test cases will be generated such that the binary tree is valid.*

---

### **Examples**

**Example 1:**

* **Input:** `descriptions = [[20,15,1], [20,17,0], [50,20,1], [50,80,0], [80,19,1]]`
* **Output:** `[50,20,80,15,17,19]`
* **Explanation:** The root node is the node with value 50 since it has no parent.

**Example 2:**

* **Input:** `descriptions = [[1,2,1], [2,3,0], [3,4,1]]`
* **Output:** `[1,2,null,null,3,4]`
* **Explanation:** The root node is the node with value 1 since it has no parent.

---

### **Constraints:**

* `1 <= descriptions.length <= 10^4`
* `descriptions[i].length == 3`
* `1 <= parent_i, child_i <= 10^5`
* `0 <= isLeft_i <= 1`
* The binary tree described by `descriptions` is guaranteed to be valid.


```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left),
 * right(right) {}
 * };
 */
class Solution {
    unordered_map<int, TreeNode*> adressMp;
    unordered_map<TreeNode*, bool> parMap;
    TreeNode* helper(vector<vector<int>>& arr) {
        for (auto vec : arr) {
            int par = vec[0];
            int child = vec[1];
            int isLeft = vec[2];
            TreeNode* parNode = nullptr;
            TreeNode* childNode = nullptr;
            if (adressMp.find(par) == adressMp.end()) {
                parNode = new TreeNode(par);
                adressMp[par] = parNode;
            } else
                parNode = adressMp[par];

            if (adressMp.find(child) == adressMp.end()) {
                childNode = new TreeNode(child);
                adressMp[child] = childNode;
            } else
                childNode = adressMp[child];
            parMap[childNode] = false;
            if (parMap.find(parNode) == parMap.end())
                parMap[parNode] = true;
            if (isLeft == 1)
                parNode->left = childNode;
            else
                parNode->right = childNode;
        }
        TreeNode* root = nullptr;
        for (auto p : parMap) {
            if (p.second == true) {
                root = p.first;
                break;
            }
        }
        return root;
    }

public:
    TreeNode* createBinaryTree(vector<vector<int>>& descriptions) {
        return helper(descriptions);
    }
};
```
