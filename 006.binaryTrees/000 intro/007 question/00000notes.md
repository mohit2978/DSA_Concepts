```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public boolean isSameTree(TreeNode p, TreeNode q) {
        if(p==null&&q==null) return true;
        if(p!=null&&q==null) return false;
        if(p==null&&q!=null) return false;
        
        boolean lres=isSameTree(p.left,q.left);
        boolean rres=isSameTree(p.right,q.right);
        
        boolean myres=(p.val==q.val);
        return lres&&rres&&myres;
        
    }
}
```

```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public List<List<Integer>> levelOrderBottom(TreeNode root) {
        List<List<Integer>> res = new ArrayList<List<Integer>>();
        if (root == null) return res;
        LinkedList <TreeNode>q=new LinkedList<>();
        q.addLast(root);
        while(q.size()>0){
            List<Integer>temp=new ArrayList<>();
            int sz=q.size();
            while(sz-->0){
                TreeNode fr=q.removeFirst();
                temp.add(fr.val);
                if(fr.left!=null)
                    q.addLast(fr.left);
                if(fr.right!=null)
                    q.addLast(fr.right);
            }
        res.add(temp);    
        }
        Collections.reverse(res);
        return res;
        
    }
}
```

Brute --> calculate height at each node O($N^2$) and then check if the node is balanced or not

better--> store height in map O(N) but 2 pass now we need in one pass

```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int data;
 *     TreeNode *left;
 *     TreeNode *right;
 *      TreeNode(int val) : data(val) , left(nullptr) , right(nullptr) {}
 * };
 **/

class Solution{
    unordered_map<TreeNode *,int> map;
    int height(TreeNode * node){
        if(node==nullptr) return 0;
        return map[node]=max(height(node->left),height(node->right))+1;
    }
    bool checkBalanced(TreeNode * node){
        if(node==nullptr) return true;
        int htdiff=abs(map[node->left]-map[node->right]);
        if(htdiff>1) return false;

        return checkBalanced(node->left) & checkBalanced(node->right);
    }
public:
    bool isBalanced(TreeNode *root){
    	if (root==nullptr) return true;
        map.clear();
        height(root);
        return checkBalanced(root);
    }
};
```

 ```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    public class nodepair{
        TreeNode child;
        int par;
        nodepair(){
            par=-1;
            }
        nodepair(TreeNode child,int par){
            this.child=child;
            this.par=par;
        }
    }
    public boolean isCousins(TreeNode root, int x, int y) {
         LinkedList <nodepair>q=new LinkedList<>();
        nodepair rt=new nodepair(root,-1);
        q.addLast(rt);
        while(q.size()>0){
            int sz=q.size();
            nodepair xchck=new nodepair();
            nodepair ychck=new nodepair();
            while(sz-->0)
            {
                nodepair fr=q.removeFirst();
                if(fr.child!=null&&fr.child.val==x){
                    xchck.child=fr.child;
                    xchck.par=fr.par;
                    if(ychck.par!=-1){
                       return ychck.par!=xchck.par;
                    }
                }
                if(fr.child!=null&&fr.child.val==y){
                    ychck.child=fr.child;
                    ychck.par=fr.par;
                    if(xchck.par!=-1){
                       return ychck.par!=xchck.par;
                    }
                }
                if(fr.child!=null&&fr.child.left!=null){
                    nodepair lc=new nodepair(fr.child.left,fr.child.val);
                    q.addLast(lc);
                }
                    
                if(fr.child!=null&&fr.child.right!=null)
                { nodepair rc=new nodepair(fr.child.right,fr.child.val);
                    q.addLast(rc);
                    
                }
                
            }

        }
        return false;
    }
}
```

  ```java
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
class Solution {
    int res=(int)-1e6;
    private void helper(TreeNode root,int mn,int mx){
        if(root==null) return;
        int t1=Math.abs(mn-root.val);
        int t2=Math.abs(mx-root.val);
        res=Math.max(t1,Math.max(t2,res));
        mn=Math.min(mn,root.val);
        mx=Math.max(mx,root.val);
        helper(root.left,mn,mx);
        helper(root.right,mn,mx);
        
    }
    public int maxAncestorDiff(TreeNode root) {
        int mn=root.val;
        int mx=root.val;
        helper(root,mn,mx);
        return res;
    }
}
  ```
