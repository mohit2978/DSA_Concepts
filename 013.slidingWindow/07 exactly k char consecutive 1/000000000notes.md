class Solution {
public:
    int longestOnes(vector<int>& nums, int k) {
        int n=nums.size();
        int si=0,ei=0,count=0,len=0;
        while(ei<n){
            if(nums[ei++]==0) count++;
            if(count>k){
                if(nums[si]==0) count--;
                si++;
            }
            len=max(len,ei-si);
        }
        return len;
    }
};
