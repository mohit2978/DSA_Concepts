#include <iostream>
#include <set>
#include <unordered_set>

using namespace std;

void showOrderedSet() {
    cout << "--- std::set (Ordered) ---" << endl;
    cout << "Underlying Data Structure: Red-Black Tree (Balanced BST)" << endl;
    
    // 1. Initialization
    set<int> s;

    // 2. Insert (TC: O(log N))
    s.insert(10);
    s.insert(5);
    s.insert(20);
    s.insert(5); // Duplicate ignored
    cout << "Elements after insertion: ";
    for(int x : s) cout << x << " "; // Output is sorted: 5 10 20
    cout << endl;

    // 3. Check Existence / Count (TC: O(log N))
    if(s.count(10)) cout << "10 is present." << endl;

    // 4. Find (TC: O(log N))
    auto it = s.find(20);
    if(it != s.end()) cout << "Found 20 using iterator." << endl;

    // 5. Erase by Value (TC: O(log N))
    s.erase(5);
    cout << "After erasing 5: ";
    for(int x : s) cout << x << " "; 
    cout << endl;

    // 6. Bounds (Unique to Ordered Set) (TC: O(log N))
    // Re-populating for demo
    s.insert(30); s.insert(40); s.insert(50);
    // Set: {10, 20, 30, 40, 50}

    auto lb = s.lower_bound(30); // First element >= 30
    auto ub = s.upper_bound(30); // First element > 30

    cout << "Lower Bound of 30: " << *lb << endl; // 30
    cout << "Upper Bound of 30: " << *ub << endl; // 40

    // 7. Size and Empty (TC: O(1))
    cout << "Size: " << s.size() << endl;
    cout << "Is Empty: " << (s.empty() ? "Yes" : "No") << endl;

    // 8. Clear (TC: O(N))
    s.clear();
    cout << "Size after clear: " << s.size() << endl << endl;
}

void showUnorderedSet() {
    cout << "--- std::unordered_set (Unordered) ---" << endl;
    cout << "Underlying Data Structure: Hash Table" << endl;

    // 1. Initialization
    unordered_set<int> us;

    // 2. Insert (Average TC: O(1), Worst Case: O(N))
    us.insert(10);
    us.insert(5);
    us.insert(20);
    us.insert(5); // Duplicate ignored
    
    cout << "Elements after insertion (Random Order): ";
    for(int x : us) cout << x << " "; 
    cout << endl;

    // 3. Check Existence / Count (Average TC: O(1))
    if(us.count(10)) cout << "10 is present." << endl;

    // 4. Find (Average TC: O(1))
    auto it = us.find(20);
    if(it != us.end()) cout << "Found 20." << endl;

    // 5. Erase (Average TC: O(1))
    us.erase(5);
    cout << "After erasing 5, size is: " << us.size() << endl;

    // 6. Size and Empty (TC: O(1))
    cout << "Size: " << us.size() << endl;

    // NOTE: lower_bound and upper_bound do NOT exist in unordered_set.

    // 7. Clear (TC: O(N))
    us.clear();
}

int main() {
    showOrderedSet();
    showUnorderedSet();
    return 0;
}
