### By fenwick tree

```java
class NumArray {
    class fenwickTree{
        private int[] fenwick;
        fenwickTree(int[] arr){
            fenwick=new int[arr.length+1];
            for(int i=1;i<fenwick.length;i++){
                update(i,arr[i-1]);
            }
        }
        private int rMsB(int x){
            return x&-x;
        }
        int sum(int pos){
            int res=0;
            while(pos>0){
                res+=fenwick[pos];
                pos-=rMsB(pos);
            }
            return res;
        }
        void update(int pos,int del){
            while(pos<fenwick.length){
                fenwick[pos]+=del;
                pos+=rMsB(pos);
            }
        }
    }
    fenwickTree ft;
    int[] arr;
    public NumArray(int[] nums) {
        ft=new fenwickTree(nums);
        arr=nums;
    }
    
    public void update(int index, int val) {
        int del=val-arr[index];
        ft.update(index+1,del);
        arr[index]=val;
    }
    
    public int sumRange(int left, int right) {
        int lpos=left+1;
        int rpos=right+1;
        return ft.sum(rpos)-ft.sum(lpos-1);
    }
}

/**
 * Your NumArray object will be instantiated and called as such:
 * NumArray obj = new NumArray(nums);
 * obj.update(index,val);
 * int param_2 = obj.sumRange(left,right);
 */
```

### By segment tree

```cpp
class NumArray {
    vector<int>segtree;
    int n=0;
    int sz=0;

 void buildTree(vector<int>& nums,int s,int e,int i){

    if(s==e){
        segtree[i]=nums[s];
        return;
    }

    int mid=(s+e)/2;
    buildTree(nums,s,mid,2*i+1);
    buildTree(nums,mid+1,e,2*i+2);

    segtree[i]=segtree[2*i+1]+segtree[2*i+2];

 }

 void updateTree(int idx,int val,int s,int e,int i){

    if(s==e){
        segtree[i]=val;
        return;
    }
    int mid=(s+e)/2;
    if(idx<=mid) updateTree(idx,val,s,mid,2*i+1);
    else updateTree(idx,val,mid+1,e,2*i+2);

    segtree[i]=segtree[2*i+1]+segtree[2*i+2];
 }

int getSum(int l,int r,int s,int e,int i){

    if(r<s || e<l) return 0;

    if(l<=s && e<=r) return segtree[i];

    int mid=(s+e)/2;

    return getSum(l,r,s,mid,2*i+1)+getSum(l,r,mid+1,e,2*i+2);
}

public:
    NumArray(vector<int>& nums) {
        n=nums.size();
        sz=4*n;
        segtree.resize(sz);
        buildTree(nums,0,n-1,0);
    }
    
    void update(int index, int val) {

        updateTree( index, val,0,n-1,0);
        
    }
    
    int sumRange(int left, int right) {
        return getSum(left,right,0,n-1,0);
    }
};

 * NumArray* obj = new NumArray(nums);
 * obj->update(index,val);
 * int param_2 = obj->sumRange(left,right);
 */
 ```
