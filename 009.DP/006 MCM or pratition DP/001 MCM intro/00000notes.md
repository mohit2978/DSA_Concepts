```cpp
class Solution{
	public:
		int matrixMultiplication(vector<int>& arr){
            int n=arr.size();
            vector<vector<int>> dp(n,vector<int>(n,0));
            for(int gap=1;gap<n;gap++){
                for(int i=0,j=gap;j<n;i++,j++){
                    if(i+1==j) continue;
                    int mini=1e9;
                    for(int k=i+1;k<=j-1;k++){
                        int ans=0;
                        ans+=dp[i][k];
                        ans+=dp[k][j];
                        ans+=(arr[i]*arr[k]*arr[j]);
                        mini=min(mini,ans);
                    }
                    dp[i][j]=mini;
                }
            }
            return dp[0][n-1];
    	}
};
```

and dp[k][j] be filled before dp[i][j] as here `k=i+1 to j-1` so dp[i][k] and dp[k][j] be filled before dp[i][j]

see we considered base case as i+1==j as single matrix 
 so there is no dp[i][i] vala case so thats why we run k=i+1 to j-1 and not i to j as 
 when k=i dp[i][i] willl be there and we do not have that and when k=j ,dp[j][j] we do not have that too

 rest is simple i guess we have answer at dp[0][n-1]
