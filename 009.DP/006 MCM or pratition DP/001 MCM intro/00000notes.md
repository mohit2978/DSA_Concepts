class Solution{
	public:
		int matrixMultiplication(vector<int>& arr){
            int n=arr.size();
            vector<vector<int>> dp(n,vector<int>(n,0));
            for(int gap=1;gap<n;gap++){
                for(int i=0,j=gap;j<n;i++,j++){
                    if(i+1==j) continue;
                    int mini=1e9;
                    for(int k=i+1;k<=j-1;k++){
                        int ans=0;
                        ans+=dp[i][k];
                        ans+=dp[k][j];
                        ans+=(arr[i]*arr[k]*arr[j]);
                        mini=min(mini,ans);
                    }
                    dp[i][j]=mini;
                }
            }
            return dp[0][n-1];
    	}
};

## Striver way

```cpp
#include <bits/stdc++.h>
using namespace std;

class Solution {
public:
    int matrixMultiplication(vector<int>& nums) {
        int n = nums.size();
        vector<vector<int>> dp(n, vector<int>(n, INT_MAX));

        // A single matrix needs no multiplication cost
        for (int i = 1; i < n; ++i) {
            dp[i][i] = 0;
        }

        // Filling the dp array
        for (int length = 2; length < n; ++length) { // length of the chain
            for (int i = 1; i <= n - length; ++i) {
                int j = i + length - 1;
                for (int k = i; k < j; ++k) {
                    int cost = dp[i][k] + dp[k + 1][j] + nums[i - 1] * nums[k] * nums[j];
                    if (cost < dp[i][j]) {
                        dp[i][j] = cost;
                    }
                }
            }
        }

        // The result is in dp[1][n-1]
        return dp[1][n - 1];
    }
};

int main() {
    Solution sol;
    vector<int> nums = {10, 15, 20, 25};
    // Output should be 8000
    cout << sol.matrixMultiplication(nums) << endl; 
    return 0;
}

```
### 1. Striver way (Standard Textbook Approach)
This code follows the classic **CLRS (Introduction to Algorithms)** textbook style.

* **Mental Model:** `dp[i][j]` means the cost to multiply **Matrix $i$** through **Matrix $j$**.
* **Indexing:** It uses **1-based indexing** for matrices.
    * Matrix 1 is dimensions `nums[0]` $\times$ `nums[1]`.
    * Matrix $i$ is dimensions `nums[i-1]` $\times$ `nums[i]`.
* **The Cost Formula:**
    `nums[i-1] * nums[k] * nums[j]`
    * `nums[i-1]`: Rows of first matrix ($A$).
    * `nums[k]`: Common dimension (Cols of left part / Rows of right part).
    * `nums[j]`: Cols of last matrix ($B$).
* **Why it looks weird:** It starts loops from `i=1` and returns `dp[1][n-1]`.

---

### 2. Rajneesh way(Gap/Interval Approach)
This code treats the indices $i$ and $j$ as **indexes in the nums array itself**, not as "Matrix ID numbers."

* **Mental Model:** `dp[i][j]` means the cost to multiply the chain of matrices defined by the dimensions from **index $i$ to index $j$** in the array.
* **The Split:** It cuts the array at index $k$.
    * **Left sub-problem:** `dp[i][k]` (Range from $i$ to $k$)
    * **Right sub-problem:** `dp[k][j]` (Range from $k$ to $j$)
* **The Cost Formula:**
    `arr[i] * arr[k] * arr[j]`
    * This is visually cleaner: The outer dimensions are `arr[i]` and `arr[j]`, and the cut is at `arr[k]`.
