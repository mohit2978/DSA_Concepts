public:
    ArrayStack() {
    }
    
    void push(int x) {
        v.push_back(x);

## Simple queue implemantation

```cpp
class ArrayQueue {
    vector<int>v;
    int stIdx=0;
    int sz=0;
public:
    ArrayQueue() {
    }
    
    void push(int x) {
        v.push_back(x);
        sz++;
    }
    
    int pop() {
        if(isEmpty()==true) return -1;
        int val=this->peek();
        stIdx++;
        sz--;
        return val;
    }
    
    int peek() {
    if(isEmpty()==true) return -1;
    return v[stIdx];
    }
    
    bool isEmpty() {
        return sz==0;
    }
};
```
