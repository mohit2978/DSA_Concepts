class Solution
{
public:
    int celebrity(vector<vector<int>> &M){
        int n=M.size();
        int i=0;
        int j=n-1;
        while(i<j){
            if(M[i][j]==0) j--;
            else i++;
        }
        for(int k=0;k<n;k++){
            if(k!=i && (M[i][k]==1 || M[k][i]==0)) return -1;
        }
        return i;
    }
};
